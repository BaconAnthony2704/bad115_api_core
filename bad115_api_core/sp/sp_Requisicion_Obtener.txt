CREATE PROCEDURE [dbo].sp_RequisicionProducto_Obtener
	@id_requisicion int = 0
AS
	begin
		select 
	re.IDREQUISICION
	,re.USUARIOENCARGADO
	,re.FECHAINGRESADA
	,re.ESTADO ESTADO_REQUISICION
	,det.CANTIDAD
	,su.NOMBRE NOMBRE_SUCURSAL
	,p.NOMBRE PAIS
	,pro.ID_PRODUCTO
	,ac.DESCRIPCION DESCRIPCION_PRODUCTO
	,pro.NOMBRE NOMBRE_PRODUCTO
	,pro.PRECIO_MINIMO
	,pro.FECHA_VENCIMIENTO
	,pro.ES_EXPORTADO
	,pro.PRECIO_MINIMO
	,pro.MODIFICADO_POR
	,pro.MODIFICADOR_EN
	,pro.ACTIVO
	from REQUISICION re
	inner join DETALLEREQUISICION det on det.IDREQUISICION=re.IDREQUISICION
	inner join PRODUCTO pro on pro.ID_PRODUCTO=det.ID_PRODUCTO
	inner join ADMIN_CATALOGO ac on ac.ID_ADMIN_CATALOGO=pro.ID_ADMIN_CATALOGO
	inner join SUCURSAL su on su.ID_SUCURSAL=re.ID_SUCURSAL
	inner join PAIS p on p.ID_PAIS=su.ID_PAIS
	where re.IDREQUISICION=@id_requisicion
	end
