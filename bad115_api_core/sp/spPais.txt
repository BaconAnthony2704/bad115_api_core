CREATE PROCEDURE spPais_ver
@p_id AS BIGINT=NULL
AS
BEGIN
	IF EXISTS(SELECT * FROM PAIS WHERE ID_PAIS=@p_id)
		SELECT * FROM PAIS WHERE ID_PAIS=@p_id
	ELSE
		SELECT * FROM PAIS
END

CREATE PROCEDURE spPais_guardar
	@p_ACTIVO AS BIT      ,
	@p_CODIGO AS VARCHAR  ( 25 ) = NULL ,
	@p_ID_PAIS AS INT      ,
	@p_MODIFICADO_EN AS DATETIME      ,
	@p_MODIFICADO_POR AS INT      ,
	@p_NOMBRE AS VARCHAR  ( 250 ) = NULL 
AS
BEGIN

	 IF (@p_ID_PAIS=-1)
	BEGIN
		INSERT INTO PAIS (
		 ACTIVO ,
		 CODIGO ,
		 MODIFICADO_EN ,
		 MODIFICADO_POR ,
		 NOMBRE 
		) VALUES (
		 @p_ACTIVO, 
		 @p_CODIGO, 
		 @p_MODIFICADO_EN, 
		 @p_MODIFICADO_POR, 
		 @p_NOMBRE 
		)
	END

	ELSE
	BEGIN
		UPDATE PAIS 
		SET 
		ACTIVO=@p_ACTIVO,
		CODIGO=@p_CODIGO,
		MODIFICADO_EN=@p_MODIFICADO_EN,
		MODIFICADO_POR=@p_MODIFICADO_POR,
		NOMBRE=@p_NOMBRE
		WHERE ID_PAIS=@p_ID_PAIS
	END
END
